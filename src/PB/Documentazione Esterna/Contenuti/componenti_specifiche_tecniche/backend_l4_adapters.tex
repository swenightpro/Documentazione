\subsubsection{Layer 5 - Infrastructure Adapters}

\paragraph{PostgresAdapter}
\textbf{Descrizione della classe:} implementa le porte repository interagendo con PostgreSQL e traducendo query applicative in operazioni SQL.

\textbf{Descrizione degli attributi:}
\begin{itemize}
    \item \texttt{connectionString}: stringa di connessione al database.
\end{itemize}

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{executeSQL(query: String)}: esegue query verso DB e restituisce il risultato elaborabile dagli adapter superiori.
\end{itemize}

\paragraph{OpenAIAdapter}
\textbf{Descrizione della classe:} realizza il contratto \texttt{IAIClient} mappando chiamate di dominio verso le API OpenAI.

\textbf{Descrizione degli attributi:}
\begin{itemize}
    \item \texttt{apiKey}: chiave di autenticazione verso il servizio OpenAI.
\end{itemize}

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{transcribeAudio(audio: File)}: invoca il modello speech-to-text.
    \item \texttt{interpretIntent(text: String, context: String)}: invia prompt contestualizzato al modello conversazionale.
    \item \texttt{createEmbedding(text: String)}: richiede embedding per vector search.
\end{itemize}

\paragraph{FileSystemAdapter}
\textbf{Descrizione della classe:} implementa la porta di esportazione ordini scrivendo file nel canale condiviso con ERP.

\textbf{Descrizione degli attributi:}
\begin{itemize}
    \item \texttt{sharedFolderPath}: percorso logico/fisico della cartella di integrazione.
\end{itemize}

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{writeOrderAsJSON(order: Order)}: serializza e salva l'ordine in JSON.
\end{itemize}

\paragraph{WebSocketAdapter}
\textbf{Descrizione della classe:} concretizza la porta di notifica realtime tramite canale WebSocket.

\textbf{Descrizione degli attributi:} il diagramma non esplicita attributi persistenti per questa classe.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{notifyOperator(operatorId: String, ticket: Ticket)}: invia notifiche operative ad un operatore specifico.
    \item \texttt{sendRealtimeUpdate(chatsessionId: String, payload: String)}: propaga aggiornamenti in tempo reale alla sessione client.
\end{itemize}
