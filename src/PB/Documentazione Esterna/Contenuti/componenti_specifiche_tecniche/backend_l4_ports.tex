\subsubsection{Outbound Ports (Interfaces)}

\paragraph{IUserRepository}
\textbf{Descrizione della classe/interfaccia:} definisce il contratto di accesso ai dati utente, credenziali e informazioni cliente.

\vspace{\baselineskip}\noindent\textbf{Descrizione degli attributi:} non applicabile (interfaccia senza stato).

\vspace{\baselineskip}\noindent\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{find\_by\_email(email: str)}: recupera l'utente associato all'email.
    \item \texttt{find\_by\_id(user\_id: int)}: recupera l'utente per ID.
    \item \texttt{update\_password(user\_id, password\_hash, password\_salt)}: aggiorna hash e salt della password.
    \item \texttt{get\_client\_info(cod\_cli: int)}: recupera ragione sociale e dati del cliente.
    \item \texttt{search\_clients(query: str, limit: int)}: cerca clienti per ragione sociale o codice.
    \item \texttt{get\_export\_folder(user\_id: int)}: recupera la cartella di esportazione.
    \item \texttt{set\_export\_folder(user\_id: int, path)}: salva la cartella di esportazione.
\end{itemize}

\paragraph{ISessionManager}
\textbf{Descrizione della classe/interfaccia:} astrae la persistenza di sessioni, messaggi e carrelli applicativi.

\vspace{\baselineskip}\noindent\textbf{Descrizione degli attributi:} non applicabile (interfaccia senza stato).

\vspace{\baselineskip}\noindent\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{get\_active\_session(user\_id: int)}: recupera la sessione attiva dell'utente.
    \item \texttt{create\_session(user\_id: int)}: crea una nuova sessione.
    \item \texttt{get\_messages(session\_id: int)}: recupera i messaggi di una sessione.
    \item \texttt{save\_message(session\_id, sender, content, metadata)}: salva un messaggio.
    \item \texttt{get\_cart\_by\_session(session\_id: int)}: recupera il carrello della sessione.
    \item \texttt{get\_cart(user\_id: int)}: recupera il carrello dell'utente.
    \item \texttt{add\_to\_cart(user\_id, cod\_art, qta, source, ...)}: aggiunge un articolo al carrello.
    \item \texttt{add\_to\_cart\_by\_session(session\_id, cod\_art, qta, source, ...)}: aggiunge per sessione.
    \item \texttt{remove\_from\_cart(cart\_item\_id, user\_id)}: rimuove un articolo.
    \item \texttt{remove\_from\_cart\_by\_session(cart\_item\_id, session\_id)}: rimuove per sessione.
    \item \texttt{update\_cart\_quantity(cart\_item\_id, user\_id, qta, source)}: aggiorna quantit\`a.
    \item \texttt{update\_cart\_quantity\_by\_session(cart\_item\_id, session\_id, qta, source)}: aggiorna per sessione.
    \item \texttt{clear\_cart(user\_id)}: svuota il carrello dell'utente.
    \item \texttt{clear\_cart\_by\_session(session\_id)}: svuota il carrello della sessione.
\end{itemize}

\paragraph{ITicketRepository}
\textbf{Descrizione della classe/interfaccia:} astrae persistenza e query dei ticket di assistenza.

\vspace{\baselineskip}\noindent\textbf{Descrizione degli attributi:} non applicabile (interfaccia senza stato).

\vspace{\baselineskip}\noindent\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{create\_ticket(session\_id, cod\_cli)}: crea un nuovo ticket.
    \item \texttt{get\_ticket\_by\_session(session\_id, cod\_cli)}: recupera il ticket per sessione.
    \item \texttt{get\_open\_tickets()}: ritorna tutti i ticket aperti o in lavorazione.
    \item \texttt{get\_ticket\_by\_id(ticket\_id)}: recupera un ticket per ID.
    \item \texttt{lock\_ticket(ticket\_id, operator\_id)}: imposta lo stato in\_lavorazione.
    \item \texttt{unlock\_ticket(ticket\_id)}: rilascia il lock.
    \item \texttt{close\_ticket(ticket\_id)}: chiude il ticket definitivamente.
    \item \texttt{get\_platform\_usage\_overview(days)}: metriche aggregate.
    \item \texttt{get\_last\_message\_time(session\_id)}: timestamp dell'ultimo messaggio.
    \item \texttt{send\_message(session\_id, sender, content)}: salva un messaggio nella chat.
\end{itemize}

\paragraph{IAIClient}
\textbf{Descrizione della classe/interfaccia:} formalizza il contratto verso i servizi di AI generativa, speech-to-text e analisi immagini.

\vspace{\baselineskip}\noindent\textbf{Descrizione degli attributi:} non applicabile (interfaccia senza stato).

\vspace{\baselineskip}\noindent\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{transcribe\_audio(audio\_file: IO[bytes], filename: str)}: converte audio in testo.
    \item \texttt{interpret\_intent(text, context, system\_prompt, model)}: classifica intenti e produce decisione AI.
    \item \texttt{generate\_embedding(text, model)}: calcola embedding vettoriale del testo.
    \item \texttt{analyze\_image(image\_base64, conversation\_history, system\_prompt)}: analizza un'immagine con GPT-5.4 Vision.
\end{itemize}
