\subsubsection{Layer 4 - Outbound Ports (Interfaces)}

\paragraph{IUserRepository}
\textbf{Descrizione della classe/interfaccia:} definisce il contratto di accesso ai dati utente e alle regole di assortimento.

\textbf{Descrizione degli attributi:} non applicabile (interfaccia senza stato).

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{findByEmail(email: String)}: recupera il cliente associato all'email.
    \item \texttt{getAssortment(clienteId: String)}: restituisce i codici prodotto autorizzati per cliente.
\end{itemize}

\paragraph{ISessionManager}
\textbf{Descrizione della classe/interfaccia:} astrae la persistenza di sessioni e carrelli applicativi.

\textbf{Descrizione degli attributi:} non applicabile (interfaccia senza stato).

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{saveSession(session: SessionData)}: persiste lo stato di sessione.
    \item \texttt{getCart(chatsessionId: String)}: recupera il carrello della sessione.
    \item \texttt{saveCart(cart: Carrello)}: aggiorna il carrello persistito.
\end{itemize}

\paragraph{IOrderRepository}
\textbf{Descrizione della classe/interfaccia:} definisce le operazioni di persistenza e ricerca ordini.

\textbf{Descrizione degli attributi:} non applicabile (interfaccia senza stato).

\textbf{Descrizione dei metodi:}
\begin{itemize}

    \item \texttt{findByCliente(clienteId: String)}: recupera ordini del cliente.
    \item \texttt{findById(orderId: String)}: recupera ordine per identificativo.
    \item \texttt{getOrders(filters, page, pageSize)}: ricerca ordini con filtri e paginazione.
\end{itemize}
