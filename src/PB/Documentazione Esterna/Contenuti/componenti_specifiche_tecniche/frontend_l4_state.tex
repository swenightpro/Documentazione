\subsubsection{Layer 2 - State Management Layer (React Context + Custom Hooks)}

\paragraph{AuthContext}
\textbf{Descrizione del componente:} gestisce stato di autenticazione lato client (token, identita e ruolo) e rappresenta la base delle guardie di accesso.

\textbf{Descrizione degli attributi:}
\begin{itemize}
    \item \texttt{token}: token JWT della sessione utente.
    \item \texttt{role}: ruolo applicativo corrente (Cliente o Operatore).
    \item \texttt{userId}: identificativo dell'utente autenticato.
\end{itemize}

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{login(payload: AuthPayload)}: inizializza lo stato di autenticazione locale.
    \item \texttt{logout()}: azzera lo stato sessione lato client.
    \item \texttt{isAuthenticated()}: verifica se l'utente e autenticato.
\end{itemize}

\paragraph{CartContext}
\textbf{Descrizione del componente:} mantiene lo stato del carrello di sessione, inclusa la confidenza AI esposta in UI operatore.

\textbf{Descrizione degli attributi:}
\begin{itemize}
    \item \texttt{sessionId}: identificativo della sessione chat associata al carrello.
    \item \texttt{items}: righe prodotto attualmente presenti nel carrello.
    \item \texttt{confidenceScore}: metrica di affidabilita AI aggregata/di contesto.
\end{itemize}

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{upsertItem(item: CartItem)}: aggiunge o aggiorna una riga prodotto nel carrello.
    \item \texttt{removeItem(productId: string)}: rimuove una riga prodotto.
    \item \texttt{clearCart()}: svuota il carrello locale a fine flusso/reset.
\end{itemize}

\paragraph{ChatContext}
\textbf{Descrizione del componente:} mantiene lo stato realtime della conversazione (messaggi, connessione, modalita AI/Operatore).

\textbf{Descrizione degli attributi:}
\begin{itemize}
    \item \texttt{messages}: storico messaggi della sessione attiva.
    \item \texttt{connectionState}: stato del canale realtime (connesso/disconnesso/reconnecting).
    \item \texttt{mode}: modalita conversazionale corrente (AI oppure Operatore).
\end{itemize}

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{appendMessage(message: ChatMessage)}: aggiunge un messaggio al feed corrente.
    \item \texttt{setMode(mode: ChatMode)}: aggiorna la modalita della sessione chat.
    \item \texttt{resetSession()}: azzera il contesto conversazionale lato frontend.
\end{itemize}

\paragraph{NotificationContext}
\textbf{Descrizione del componente:} centralizza notifiche transitorie e alert applicativi provenienti da azioni utente o eventi realtime.

\textbf{Descrizione degli attributi:}
\begin{itemize}
    \item \texttt{queue}: coda notifiche visualizzabili nella UI.
\end{itemize}

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{push(notification: NotificationDTO)}: inserisce una nuova notifica in coda.
    \item \texttt{dismiss(notificationId: string)}: rimuove una notifica specifica.
    \item \texttt{clear()}: svuota tutte le notifiche pendenti.
\end{itemize}

\paragraph{CustomHooks}
\textbf{Descrizione del componente:} fornisce API di accesso uniforme ai contesti e agli stati applicativi, riducendo duplicazione logica nei componenti UI.

\textbf{Descrizione degli attributi:} non sono esplicitati attributi nel diagramma.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{useAuth()}: espone stato e operazioni di autenticazione.
    \item \texttt{useCart()}: espone stato e operazioni di carrello.
    \item \texttt{useChat()}: espone stato e operazioni di chat.
    \item \texttt{useTickets()}: espone stato ticket per dashboard operatore.
    \item \texttt{useOrders()}: espone stato ordini per viste storico/globali.
    \item \texttt{useAILogs()}: espone stato AI log per pipeline HITL.
    \item \texttt{useRealtime()}: espone stato canale realtime e sottoscrizioni.
\end{itemize}

\paragraph{RoleBasedRendering}
\textbf{Descrizione del componente:} applica regole di rendering condizionale in base al ruolo, assicurando che le viste operatore non siano esposte ai clienti.

\textbf{Descrizione degli attributi:} nessun attributo di stato esplicitato nel diagramma.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{canRender(role: UserRole, feature: string)}: valuta la visibilita di una feature per un ruolo.
    \item \texttt{renderForRole(role: UserRole)}: compone la porzione UI consentita dal ruolo corrente.
\end{itemize}
