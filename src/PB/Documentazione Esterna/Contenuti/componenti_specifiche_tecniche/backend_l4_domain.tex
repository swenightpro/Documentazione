\subsubsection{Core Domain (Entities \& Value Objects)}

\paragraph{Cliente}
\vspace{\baselineskip}\noindent\textbf{Descrizione della classe:} rappresenta il cliente commerciale autenticato e abilitato all'inserimento ordini.

\vspace{\baselineskip}\noindent\textbf{Descrizione degli attributi:}
\begin{itemize}
    \item \texttt{id}: identificativo univoco (usato per join interne).
    \item \texttt{rag\_soc}: denominazione legale dell'azienda cliente.
    \item \texttt{email}: contatto email principale per autenticazione/comunicazioni.
\end{itemize}

\vspace{\baselineskip}\noindent\textbf{Descrizione dei metodi:} non sono definiti metodi nel diagramma per questa entita.

\paragraph{Operatore}
\vspace{\baselineskip}\noindent\textbf{Descrizione della classe:} identifica l'utente interno che prende in carico ticket e supervisiona i casi critici.

    \item \texttt{id}: identificativo univoco operatore.
    \item \texttt{nome}: nominativo operatore.

\paragraph{Order}
\vspace{\baselineskip}\noindent\textbf{Descrizione della classe:} rappresenta l'ordine consolidato pronto per storicizzazione ed esportazione verso ERP.

\vspace{\baselineskip}\noindent\textbf{Descrizione degli attributi:}
\begin{itemize}
    \item \texttt{id}: identificativo ordine.
    \item \texttt{cod\_cli}: riferimento al cliente proprietario dell'ordine.
    \item \texttt{created\_at}: timestamp di emissione.
    \item \texttt{righe}: collezione delle righe ordine derivate dal carrello.
\end{itemize}

\vspace{\baselineskip}\noindent\textbf{Descrizione dei metodi:} nel diagramma non sono mostrati metodi per questa entita.

    \centering
    \includegraphics[width=0.52\textwidth]{uml_backend/24_order.png}
    \caption{UML della classe \texttt{Order}.}
\end{figure}

\paragraph{Carrello}
\vspace{\baselineskip}\noindent\textbf{Descrizione della classe:} modella lo stato temporaneo dell'ordine in costruzione per una sessione chat.

\vspace{\baselineskip}\noindent\textbf{Descrizione degli attributi:}
\begin{itemize}
    \item \texttt{session\_id}: identificativo sessione conversazionale associata.
    \item \texttt{cod\_cli}: cliente a cui il carrello appartiene.
    \item \texttt{righe}: elenco righe prodotto correnti.
\end{itemize}

\vspace{\baselineskip}\noindent\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{calcolaTotalePezzi()}: restituisce la somma totale delle quantita nel carrello.
\end{itemize}

\paragraph{LogAI}
\vspace{\baselineskip}\noindent\textbf{Descrizione della classe:} traccia input, output e correzioni legate alle elaborazioni AI per audit e miglioramento modello.

\vspace{\baselineskip}\noindent\textbf{Descrizione degli attributi:}
\begin{itemize}
    \item \texttt{id}: identificativo del log.
    \item \texttt{inputOriginario}: testo/audio trascritto ricevuto dal cliente.
    \item \texttt{rispostaLLM}: proposta generata dal modello linguistico.
    \item \texttt{correzioneDelta}: differenza applicata manualmente da utente/operatore.
    \item \texttt{stato}: stato del log nel workflow di validazione.
\end{itemize}
