\subsubsection{Controllers (Inbound Ports)}

\paragraph{AuthController}
\vspace{\baselineskip}\noindent\textbf{Descrizione della classe:} gestisce gli endpoint di autenticazione, cambio password, gestione cartella di esportazione e rilascio token SSE, traducendo le richieste HTTP in invocazioni ai casi d'uso applicativi.

\vspace{\baselineskip}\noindent\textbf{Descrizione degli attributi:} nessun attributo di stato esplicitato nel diagramma (controller stateless).

\vspace{\baselineskip}\noindent\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{login(body: LoginRequest)}: avvia autenticazione utente e genera la sessione applicativa tramite cookie JWT.
    \item \texttt{logout()}: invalida la sessione autenticata corrente cancellando i cookie.
    \item \texttt{me()}: restituisce i dati dell'utente correntemente autenticato.
    \item \texttt{change\_password(body: ChangePasswordRequest)}: gestisce il flusso di aggiornamento credenziali.
    \item \texttt{get\_export\_folder()}: recupera la cartella di esportazione configurata dall'operatore.
    \item \texttt{set\_export\_folder(body)}: imposta o aggiorna la cartella di esportazione dell'operatore.
    \item \texttt{sse\_token()}: restituisce il JWT dal cookie HTTPOnly per connessioni SSE dirette.
\end{itemize}

\paragraph{ConversationController}
\vspace{\baselineskip}\noindent\textbf{Descrizione della classe:} espone gli endpoint per la conversazione multimodale (testo, audio, immagini) e il recupero dello storico chat e sessioni.

\vspace{\baselineskip}\noindent\textbf{Descrizione degli attributi:} nessun attributo di stato esplicitato nel diagramma (controller stateless).

\vspace{\baselineskip}\noindent\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{chat(body: ChatRequest)}: elabora input testuale inviato dall'utente tramite la pipeline AI.
    \item \texttt{chat\_image(file: UploadFile)}: gestisce input multimodale includendo immagini con OCR via GPT-5.4 Vision.
    \item \texttt{save\_message\_to\_session(body: SaveMessageRequest)}: salva un messaggio UI nella sessione per persistenza.
    \item \texttt{transcribe(file: UploadFile)}: effettua trascrizione di messaggi vocali via Whisper.
    \item \texttt{get\_active\_session()}: recupera la sessione conversazionale attiva dell'utente.
    \item \texttt{create\_session()}: istanzia una nuova sessione conversazionale.
    \item \texttt{get\_messages(session\_id: int)}: recupera lo storico messaggi della sessione.
\end{itemize}

\paragraph{CartController}
\vspace{\baselineskip}\noindent\textbf{Descrizione della classe:} governa le operazioni sul carrello (inserimento, modifica, rimozione) e l'aggiunta per nome prodotto.

\vspace{\baselineskip}\noindent\textbf{Descrizione degli attributi:} nessun attributo di stato esplicitato nel diagramma (controller stateless).

\vspace{\baselineskip}\noindent\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{get\_cart(session\_id: int)}: restituisce l'elenco corrente delle righe prodotto nel carrello.
    \item \texttt{cart\_action(body: CartActionRequest)}: agisce centralmente per inserire, modificare quantit\`a o rimuovere articoli dal carrello.
    \item \texttt{add\_by\_name(body: AddByNameRequest)}: aggiunge un prodotto al carrello ricercandolo per nome testuale.
\end{itemize}

\paragraph{OrderController}
\vspace{\baselineskip}\noindent\textbf{Descrizione della classe:} espone le API per creazione ordini, consultazione storico, dettaglio singolo ordine, esportazione singola e batch.

\vspace{\baselineskip}\noindent\textbf{Descrizione degli attributi:} nessun attributo di stato esplicitato nel diagramma (controller stateless).

\vspace{\baselineskip}\noindent\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{create\_order(body: CreateOrderRequest)}: tramuta un carrello finale in ordine persistente.
    \item \texttt{list\_orders(...)}: recupera lo storico ordini paginato per il cliente loggato.
    \item \texttt{list\_all\_orders(...)}: interfaccia estesa a disposizione dell'operatore con filtri avanzati.
    \item \texttt{order\_detail(id: int)}: recupera il dettaglio completo di un ordine specifico.
    \item \texttt{export\_order(id: int, format: str)}: esportazione dell'ordine singolo in formato JSON o CSV.
    \item \texttt{export\_batch\_orders(...)}: avvia macroesportazione filtrata di ordini multipli.
\end{itemize}

\paragraph{TicketController}
\vspace{\baselineskip}\noindent\textbf{Descrizione della classe:} gestisce l'apertura, la presa in carico, la chiusura e la messaggistica dei ticket di assistenza Human-in-the-Loop, sia lato operatore che lato cliente.

\vspace{\baselineskip}\noindent\textbf{Descrizione degli attributi:} nessun attributo di stato esplicitato nel diagramma (controller stateless).

\vspace{\baselineskip}\noindent\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{list\_open\_tickets()}: restituisce la lista dei ticket pendenti (operatori).
    \item \texttt{create\_ticket(body: CreateTicketRequest)}: crea un ticket di assistenza per la sessione corrente.
    \item \texttt{get\_ticket\_by\_session(session\_id: int)}: ritorna il ticket associato a una sessione.
    \item \texttt{get\_analytics\_overview(days: int)}: metriche aggregate per la dashboard operatore.
    \item \texttt{get\_ticket(ticket\_id: int)}: dettaglio di un ticket specifico.
    \item \texttt{lock\_ticket(ticket\_id: int)}: assegna il ticket all'operatore corrente (lock).
    \item \texttt{send\_ticket\_message(ticket\_id: int, body: SendMessageRequest)}: invia un messaggio operatore nella chat del ticket.
    \item \texttt{unlock\_ticket(ticket\_id: int)}: rilascia il lock sull'operatore.
    \item \texttt{close\_ticket(ticket\_id: int)}: chiude un ticket (operatore o cliente).
    \item \texttt{close\_ticket\_by\_session(session\_id: int)}: chiude il ticket associato a una sessione (cliente).
    \item \texttt{send\_customer\_message(session\_id: int, body: SendMessageRequest)}: invia un messaggio del cliente nella chat del ticket.
\end{itemize}
