\subsubsection{Layer 3 - Service Layer (API Client Modules REST/HTTPS + JWT)}

\paragraph{authService}
\textbf{Descrizione del componente:} facade di accesso alle API di autenticazione e gestione credenziali.

\textbf{Descrizione degli attributi:} il diagramma non esplicita attributi persistenti.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{login(credentials: CredentialsDTO)}: esegue autenticazione e recupera sessione/token.
    \item \texttt{logout()}: invalida la sessione lato backend.
    \item \texttt{resetPassword(req: ResetPasswordDTO)}: avvia procedura di reset password.
    \item \texttt{changePassword(req: PasswordChangeDTO)}: aggiorna la password dell'utente autenticato.
\end{itemize}

\paragraph{cartService}
\textbf{Descrizione del componente:} modulo API dedicato alle operazioni di carrello e invio ordine, separato dal modulo di sola ricerca catalogo.

\textbf{Descrizione degli attributi:} non sono esplicitati attributi nel diagramma.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{getCart(sessionId: string)}: recupera lo stato corrente del carrello.
    \item \texttt{upsertCartItem(req: CartItemUpsertDTO)}: inserisce/aggiorna una riga carrello.
    \item \texttt{removeItem(sessionId: string, productId: string)}: rimuove un articolo dal carrello.
    \item \texttt{submitOrder(sessionId: string)}: conferma il checkout e invia l'ordine al backend.
\end{itemize}

\paragraph{orderService}
\textbf{Descrizione del componente:} modulo di consultazione ordini gia finalizzati (storico, dettaglio, export CSV).

\paragraph{ticketService}
\textbf{Descrizione del componente:} modulo API di supporto Human-in-the-Loop per gestione ticket cliente-operatore.

\textbf{Descrizione degli attributi:} non sono rappresentati attributi di stato nel diagramma.

\textbf{Descrizione dei metodi:}
\begin{itemize}

    \item \texttt{openTicket(req: OpenTicketDTO)}: apre un nuovo ticket di assistenza.
    \item \texttt{closeTicket(ticketId: string)}: chiude un ticket in carico.

\paragraph{feedbackService}
\textbf{Descrizione del componente:} modulo API dedicato al workflow di validazione delle discrepanze AI in dashboard operatore.

\textbf{Descrizione degli attributi:} nessun attributo di stato esplicitato nel diagramma.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{getLogs(filters: AILogFilterDTO)}: recupera lista log AI filtrata.
    \item \texttt{approveSig(logId: string)}: approva una segnalazione per pipeline HITL.
    \item \texttt{rejectSig(logId: string)}: scarta una segnalazione non valida.
    \item \texttt{revokeSig(logId: string)}: revoca un'approvazione precedente.
\end{itemize}
