\subsubsection{Layer 3 - Service Layer (API Client Modules REST/HTTPS + JWT)}

\paragraph{authService}
\textbf{Descrizione del componente:} facade di accesso alle API di autenticazione e gestione credenziali.

\textbf{Descrizione degli attributi:} il diagramma non esplicita attributi persistenti.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{login(email: string, password: string)}: esegue autenticazione e recupera sessione/token.
    \item \texttt{getProfile()}: recupera i dati dell'utente autenticato.
    \item \texttt{changePassword(...)}: aggiorna la password dell'utente autenticato.
    \item \texttt{logout()}: invalida la sessione lato backend.
    \item \texttt{getExportFolder()}: recupera la directory di esportazione dell'operatore.
    \item \texttt{setExportFolder(path)}: imposta la directory di esportazione.
\end{itemize}

\paragraph{cartService}
\textbf{Descrizione del componente:} modulo API dedicato alle operazioni di carrello, con supporto per add/remove/update per sessione.

\textbf{Descrizione degli attributi:} non sono esplicitati attributi nel diagramma.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{getCart(sessionId?)}: recupera lo stato corrente del carrello.
    \item \texttt{addItem(codArt, qta, extra?)}: inserisce una nuova riga prodotto nel carrello.
    \item \texttt{removeItem(id, sessionId?)}: rimuove un articolo dal carrello.
    \item \texttt{updateQty(id, qta, source, sessionId?)}: modifica la quantit\`a di un articolo.
    \item \texttt{addByName(productName, qta, sessionId?)}: aggiunge un prodotto cercandolo per nome.
    \item \texttt{addByCode(codArt, qta, sessionId?)}: aggiunge un prodotto per codice articolo.
\end{itemize}

\paragraph{orderService}
\textbf{Descrizione del componente:} modulo di gestione ordini: creazione, consultazione storico, dettaglio, esportazione singola e batch.

\textbf{Descrizione degli attributi:} il diagramma non esplicita attributi persistenti.

\textbf{Descrizione dei metodi:}
\begin{itemize}

    \item \texttt{list(codCli, page, filters)}: lista paginata per il frontend utente.
    \item \texttt{getDetail(orderId, codCli)}: recupera il dettaglio completo di un ordine.
    \item \texttt{create(codCli, sessionId, items)}: crea un nuovo ordine dal carrello.
    \item \texttt{exportOrder(orderId, format)}: esporta un singolo ordine in JSON o CSV.
    \item \texttt{exportBatch(filters, limit)}: esportazione batch di ordini filtrati.
\end{itemize}

\paragraph{ticketService}
\textbf{Descrizione del componente:} modulo API di supporto Human-in-the-Loop per gestione ticket cliente-operatore.

\textbf{Descrizione degli attributi:} non sono rappresentati attributi di stato nel diagramma.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{getOpenTickets()}: recupera l'elenco dei ticket aperti/in lavorazione (operatore).
    \item \texttt{getTicketDetail(ticketId)}: recupera il dettaglio di un ticket (operatore).
    \item \texttt{getPlatformUsageOverview(days)}: metriche aggregate per la dashboard operatore.
    \item \texttt{lockTicket(ticketId)}: presa in carico ticket (operatore).
    \item \texttt{unlockTicket(ticketId)}: rilascio lock senza chiusura.
    \item \texttt{closeTicket(ticketId, closedBy?)}: chiusura ticket (operatore).
    \item \texttt{openTicket(sessionId)}: apertura ticket dal lato cliente.
    \item \texttt{getTicketBySession(sessionId)}: recupera ticket corrente per sessione (cliente).
    \item \texttt{closeTicketBySession(sessionId)}: chiusura ticket dal lato cliente.
    \item \texttt{sendChatMessage(ticketId, content)}: messaggio operatore nella chat ticket.
    \item \texttt{sendCustomerMessage(sessionId, content)}: messaggio cliente nella chat ticket.
\end{itemize}

\paragraph{chatService}
\textbf{Descrizione del componente:} facade verso le API conversazionali: messaggi multimodali (testo, audio, immagine), storico e persistenza UI.
