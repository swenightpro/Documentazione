\subsubsection{Layer 2 - Application Services (Use Cases)}

\paragraph{AuthService}
\textbf{Descrizione della classe:} incapsula i casi d'uso di autenticazione e invalidazione sessione.

\textbf{Descrizione degli attributi:} non sono esplicitati attributi nel diagramma; il servizio e modellato come componente applicativo stateless.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{authenticateUser(email: String, pass: String)}: valida credenziali e produce token di autenticazione.
    \item \texttt{invalidateSession(authsessionId: String)}: disattiva una sessione autenticata.
\end{itemize}

    \caption{UML della classe \texttt{AuthService}.}
\end{figure}

\paragraph{ConversationService}
\textbf{Descrizione della classe:} implementa il flusso conversazionale (testo e audio) e la gestione del contesto dialogico.

\textbf{Descrizione degli attributi:} non sono esplicitati attributi di stato nel diagramma.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{handleText(text: String, sessionId: String)}: processa input testuale e restituisce risposta chat.
    \item \texttt{handleAudio(audioBlob: File, sessionId: String)}: gestisce input audio con trascrizione e interpretazione.
    \item \texttt{clearContext(chatsessionId: String)}: cancella lo stato conversazionale associato alla sessione.
\end{itemize}

\paragraph{CartService}
\textbf{Descrizione della classe:} concentra la logica di business del carrello e la finalizzazione ordine.

\textbf{Descrizione degli attributi:} non sono indicati attributi persistenti nel diagramma.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{manipulateCart(action: CartAction)}: applica operazioni di aggiunta/modifica/rimozione su carrello.
    \item \texttt{finalizeOrder(chatsessionId: String)}: converte il carrello in ordine definitivo.
    \item \texttt{validateAssortment(clienteId: String, prodottoId: String)}: verifica interna di validita assortimento cliente-articolo.
\end{itemize}

\paragraph{TicketService}
\textbf{Descrizione della classe:} implementa l'intero ciclo di vita del ticket di assistenza.

\textbf{Descrizione degli attributi:} nessun attributo esplicitato nel diagramma.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{createTicket(clienteId: String, reason: String)}: crea un ticket a partire da richiesta cliente.
    \item \texttt{assignOperator(ticketId: String, operatorId: String)}: assegna un operatore al ticket.
    \item \texttt{resolveTicket(ticketId: String)}: chiude il ticket come risolto.
\end{itemize}

\paragraph{FeedbackService}
\textbf{Descrizione della classe:} presidia logging delle correzioni manuali e raccolta feedback negativi verso il modulo AI.

\textbf{Descrizione degli attributi:} il diagramma non esplicita attributi interni.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{logCorrection(delta: String, chatsessionId: String)}: persiste lo scostamento tra proposta AI e correzione utente.
    \item \texttt{saveNegativeFeedback(feedback: FeedbackData)}: salva feedback negativo per analisi qualitativa.
\end{itemize}
