\subsubsection{Layer 4 - Real-time Layer (WebSocket / SSE Gateway)}

\paragraph{RTGateway}
\textbf{Descrizione del componente:} gestisce la connessione realtime persistente con backend (WebSocket/SSE), inclusi heartbeat e riconnessione automatica.

\textbf{Descrizione degli attributi:}
\begin{itemize}
    \item \texttt{socketUrl}: endpoint realtime del backend FastAPI.
    \item \texttt{heartbeatMs}: intervallo heartbeat per mantenere viva la connessione.
\end{itemize}

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{connect(token: string)}: apre la connessione autenticata verso il gateway realtime.
    \item \texttt{reconnect()}: riattiva la connessione in caso di fault di rete.
    \item \texttt{subscribe(topic: string, handler: Function)}: registra un handler per tipologia evento.
    \item \texttt{dispatch(event: RealtimeEvent)}: instrada evento in ingresso all'handler dedicato.
\end{itemize}

\paragraph{ChatEventHandler}
\textbf{Descrizione del componente:} elabora eventi chat in tempo reale e aggiorna il contesto conversazionale visualizzato all'utente.

\textbf{Descrizione degli attributi:} il diagramma non esplicita attributi persistenti.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{handle(event: ChatEvent)}: interpreta l'evento chat ricevuto dal gateway.
    \item \texttt{pushToContext(ctx: ChatContext)}: sincronizza l'evento sullo stato chat condiviso.
\end{itemize}

\paragraph{TicketEventHandler}
\textbf{Descrizione del componente:} gestisce eventi ticket (apertura, presa in carico, chiusura) e aggiorna dashboard operatore in live sync.

\textbf{Descrizione degli attributi:} nessun attributo di stato esplicitato nel diagramma.

\textbf{Descrizione dei metodi:}
\begin{itemize}
    \item \texttt{handle(event: TicketEvent)}: processa l'evento ticket proveniente dal backend.
    \item \texttt{updateTicketBoard()}: aggiorna la vista tabellare ticket senza reload pagina.
    \item \texttt{notifyOperator(notification: NotificationDTO)}: mostra notifica contestuale all'operatore.
\end{itemize}
